async function applyOptimization(){ if(!optCurrentItemId){ alert('⚠️ No hay ninguna publicación seleccionada'); return; } const newTitle = document.getElementById('opt-final-title').value.trim(); const newDesc = document.getElementById('opt-final-desc').value.trim(); const resEl = document.getElementById('opt-apply-res'); if(!newTitle){ alert('⚠️ El título no puede estar vacío'); return; } if(newTitle.length < 20 || newTitle.length > 60){ if(!confirm('⚠️ El título tiene ' + newTitle.length + ' caracteres. ML recomienda 40-60.\n¿Continuar igual?')){ return; } } resEl.innerHTML = '
Actualizando publicación en ML...
'; try{ // 1. Actualizar título const updateBody = { title: newTitle }; const r1 = await fetch(WORKER + '/api/items/' + optCurrentItemId, { method: 'PUT', headers: { 'Authorization': 'Bearer ' + token, 'Content-Type': 'application/json' }, body: JSON.stringify(updateBody) }); if(!r1.ok){ const err = await r1.json(); throw new Error(err.message || 'Error al actualizar título'); } // 2. Actualizar descripción (si hay) if(newDesc){ const descBody = { plain_text: newDesc }; const r2 = await fetch(WORKER + '/api/items/' + optCurrentItemId + '/description', { method: 'PUT', headers: { 'Authorization': 'Bearer ' + token, 'Content-Type': 'application/json' }, body: JSON.stringify(descBody) }); // La descripción puede fallar en algunos casos, no es crítico if(!r2.ok){ console.warn('Advertencia: descripción no se actualizó'); } } // 3. Actualizar en cache local const itemIndex = allItems.findIndex(i => i.id === optCurrentItemId); if(itemIndex !== -1){ allItems[itemIndex].title = newTitle; } // 4. Mostrar éxito resEl.innerHTML = `
✅ ¡Publicación actualizada en ML!
Título: ${newTitle}
${newDesc ? 'Descripción actualizada ✓' : ''}
Ver en ML →
`; // Limpiar formulario después de 3 segundos setTimeout(() => { document.getElementById('opt-modify-wrap').style.display = 'none'; document.getElementById('opt-search').value = ''; document.getElementById('opt-in').value = ''; document.getElementById('opt-desc').value = ''; document.getElementById('opt-selected').innerHTML = ''; optCurrentItemId = null; }, 3000); } catch(err){ resEl.innerHTML = `
❌ Error al actualizar: ${err.message}
Verificá que tengas permisos de escritura en la app de ML.
`; } }
